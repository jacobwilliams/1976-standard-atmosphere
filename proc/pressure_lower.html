pure real(wp) function pressure_lower(H)
    real(wp),intent(in) :: H
    integer :: i, b
    b = findb(H)
    i = b + 1
    if (Lmb(i) == 0) then
        pressure_lower = Pb(i) * exp(-g0 * M0 * (H - Hb(i)) / (Rstar * Tmb(i)))
    else
        pressure_lower = Pb(i) * (Tmb(i) / (Tmb(i) + Lmb(i) * (H - Hb(i)))) ** (g0 * M0 / (Rstar * Lmb(i)))
    end if
end function pressure_lower
