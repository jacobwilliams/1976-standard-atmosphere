pure function find(x, xvec) result(i)
    real(wp),intent(in) :: x
    real(wp),dimension(:),intent(in) :: xvec
    integer :: i
    i = 1
    do
        if (i >= size(xvec)) exit
        if (x <= xvec(i + 1)) exit
        i = i + 1
    end do
end function find
