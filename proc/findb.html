pure function findb(H) result(b)
    real(wp),intent(in) :: H
    integer :: b
    integer :: i
    i = 1
    ! do while (i < size(Hb) .and. H > Hb(i + 1))
    !     i = i + 1
    ! end do
    do 
        if (i >= size(Hb)) exit
        if (H <= Hb(i + 1)) exit
        i = i + 1
    end do
    b = i - 1
end function findb
